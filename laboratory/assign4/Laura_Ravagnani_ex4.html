# model 
signal <- function(x, a, b, x0, w, t){
	t * (a * exp(-(x-x0)^2/(2*w^2)) + b)
}

par(mfrow = c(2, 3))

for(i in 1:length(w)){
	# generate observed data
	xdat <- seq(from = -7*w[i], to = 7*w[i], by = 0.5*w[i])
	signal.true <- signal(xdat, a.real, b.real, x0, w[i], delta.t)
	distrib_dat <- rpois(length(signal.true), signal.true)

	# calculate the log of the posterior on a grid {a_i, b_i}
	Nsamp <- 100
    a <- seq(0, 4, 4/Nsamp)
    b <- seq(0.5, 1.5, 1/Nsamp)
    
	# log posterior
	log.post <- function(d, x, a, b, x0, w, t){
		if(a < 0 || b < 0) {return(-Inf)}
		sum(dpois(d, lambda = signal(x, a, b, x0, w, t), log = TRUE))
	}

	logL <- matrix(0, nrow = length(a), ncol = length(b))
	for(j in 1:length(a)){
		for(k in 1:length(b)){
			logL[j,k] <- log.post(distrib_dat, xdat, a[j], b[k], x0, w[i], delta.t)
		}
	}

    # set maximum to zero and calculate the posterior
	logL <- logL - max(logL)
    post <- exp(logL)

	contour(a, b, post,
            nlevels = 5,
            labcex = 0.5,
            xlab = "a", ylab = "b",
            main = paste("w = ", w[i]))
	abline(v = a.real,
		col = "green3")
	abline(h = b.real,
		col = "green3")
}

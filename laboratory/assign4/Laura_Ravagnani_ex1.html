The likelihood is a Poisson distribution with $\mu$ unknown $$f(\{y_i\}|\mu) \propto \mu^{\sum_i{y_i}}e^{-n\mu}$$

# function to find the median
find_median <- function(f){
	p <- 0
	integral <- sum(f)
	for(i in 1:length(f)){
		p <- p + f[i]
		if(p > 0.5 * integral){
			break
		}
	}
	return(mu[i-1])
}

# function to find the min of the credibility interval (2.5%)
min_ci <- function(f){
	val <- 0
	integral <- sum(f)
	for(i in 1:length(f)){
		val <- val + f[i]
		if(val > 0.025 * integral){
			break
		}
	}
	return(mu[i-1])
}

# function to find the max of the credibility interval (97.5%)
max_ci <- function(f){
	val <- 0
	integral <- sum(f)
	for(i in 1:length(f)){
		val <- val + f[i]
		if(val > 0.975 * integral){
			break
		}
	}
	return(mu[i-1])
}

# mean
unif_mean <- integrate(function(mu) dgamma(mu, sum_y + 1, n) * mu, 0, 10)

# median
unif_median <- find_median(unif_posterior)
jeff_median <- find_median(jeff_posterior)

# second moment and variance
unif_second <- integrate(function(mu) dgamma(mu, sum_y + 1, n) * mu^2, 0, 10) 
jeff_second <- integrate(function(mu) dgamma(mu, sum_y + 1/2, n) * mu^2, 0, 10) 

unif_variance <- unif_second$value - unif_mean$value^2
jeff_variance <- jeff_second$value - jeff_mean$value^2

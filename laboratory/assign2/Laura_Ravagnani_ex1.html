# write the pdf and the cdf
dprob <- function(x) {
    pdf <- ifelse(x >= 1 & x <= 5, x/15, 0)
    return (pdf)
}

# write the cdf
pprob <- function(x) {
    cdf <- rep(0, length(x))
        for (i in 2:length(x)) {
            cdf[i] <- cdf[i - 1] + dprob(x)[i]
        }
    return(cdf)
}

# function that samples from the pdf using the inverse method
rprob <- function(x, n){
    u <- runif(n)
    val <- rep(0, n)
    for (i in 1:n) {
        val[i] <- min(x[pprob(x) >= u[i]])
    }
    return(val)
}
